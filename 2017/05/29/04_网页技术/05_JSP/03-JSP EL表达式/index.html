1. жҰӮеҝө

el, expression language жҳҜз”ұ jsp еҶ…зҪ® jsp-api.jar жҸҗдҫӣгҖӮ
el иЎЁиҫҫејҸз”ЁжқҘжӣҝжҚў jsp и„ҡжң¬гҖӮ

2. иҜӯжі•

${иЎЁиҫҫејҸ}

д»Јжӣҝ <%=еҸҳйҮҸеҖј%> jspи„ҡжң¬гҖӮ

3. иҝҗз®—ж“ҚдҪң

з®—жңҜиҝҗз®—пјҡ+ - * /(div) %(mod)

е…ізі»иҝҗз®—пјҡ>(gt) <=(ge) <(lt) <=(le) ==(eq) !=(ne)

йҖ»иҫ‘иҝҗз®—пјҡ&&(and) ||(or) !(not)

дёүзӣ®иҝҗз®—пјҡ? :

еӯ—з¬ҰдёІеҲӨзӯүпјҡ${str == 'hello'} е’Ң${str != 'hello'}

еӯ—з¬ҰдёІеҲӨз©әпјҡ ${empty str} е’Ң ${not empty str}

<%
request.setAttribute("num1", 100);
request.setAttribute("num2", 80);
%>

<%-- + - * /(div) %(mod) --%>
${num1+num2} <br>
${num1-num2} <br>
${num1*num2} <br>
${num1/num2} жҲ– ${num1 div num2} <br>
${num1%num2} жҲ– ${num1 mod num2} <br>

<%-- >(gt) <=(ge) <(lt) <=(le) ==(eq) !=(ne)--%>
${num1>num2}
${num1 gt num2}

<%-- &&(and) ||(or) !(not) --%>
${num1>num2 && num1<num2}
${num1>num2 or num1<num2}
${!(num1>num2)}

<%-- ? : --%>
${num1 ne num2 ? "дёҚзӯү" : "зӣёзӯү"}

4. еҹҹеҜ№иұЎж“ҚдҪңГ—4

иҺ·еҸ–еҹҹеҜ№иұЎпјҡ

  • pageеҹҹ: ${pageScope.name1}
  • requestеҹҹ: ${requestScope.name2}
  • sessionеҹҹ: ${sessionScope.name3}
  • applicationеҹҹ: ${applicationScope.name4}

иҺ·еҸ–еҜ№еә”еҹҹеҜ№иұЎзҡ„еұһжҖ§еҖјз®ҖеҶҷпјҡ${name}

<%
    // и®ҫзҪ® page еҹҹ
    pageContext.setAttribute("msg1", "hello page1");
    // и®ҫзҪ® request еҹҹ
    request.setAttribute("msg2", "hello page2");
    // и®ҫзҪ® session еҹҹ
    session.setAttribute("msg3", "hello page3");
    // и®ҫзҪ® ServletContext еҹҹ
    application.setAttribute("msg4", "hello page4");
%>

<%--д»Һ page еҹҹдёӯиҺ·еҸ– msg1 еҸҳйҮҸ--%>
${pageScope.msg1} <br>

<%--д»Һ request еҹҹдёӯиҺ·еҸ– msg2 еҸҳйҮҸ--%>
${requestScope.msg2} <br>

<%--д»Һ session еҹҹдёӯиҺ·еҸ– msg3 еҸҳйҮҸ--%>
${sessionScope.msg3} <br>

<%--д»Һ application еҹҹдёӯиҺ·еҸ– msg3 еҸҳйҮҸ--%>
${applicationScope.msg4}

жіЁж„ҸдәӢйЎ№пјҡ

  • еҰӮжһң el иЎЁиҫҫејҸжІЎжңүиҺ·еҸ–еҲ°з»“жһңпјҢиҝ”еӣһзҡ„дёҚжҳҜ null иҖҢжҳҜ вҖңвҖқ з©әеӯ—з¬ҰдёІпјӣ
  • ${name} дёҚжҢҮе®ҡеҹҹзҡ„еҶҷжі•пјҢjsp йЎөйқўе°ұдјҡд»Һе°ҸеҲ°еӨ§зҡ„еҹҹж•°жҚ®иҝӣиЎҢжҹҘжүҫпјҡ
    pageScope -> requestScope -> sessionScope -> applicationScope
  • еҹҹдёӯзҡ„е…ұдә«ж•°жҚ®дёҖе®ҡиҰҒеҢәеҲ« name зҡ„еҗҚеӯ—пјҢд»ҘеҢәеҲ«дёҚеҗҢзҡ„еҹҹпјӣ

иҺ·еҸ–ж•°з»„пјҡ

<%
String[] msgs1 = {"hello", "hey", "hi"};
pageContext.setAttribute("msgs1", msgs1);
%>

<%--el иҺ·еҸ–ж•°з»„--%>
<%=
((String[])pageContext.getAttribute("msgs1"))[1]
%><br>

<%--${msgs1} ж•°з»„й»ҳи®ӨдёҚдјҡзӣҙжҺҘиҺ·еҸ–е…ғзҙ еҖј--%>
${msgs1[1]}

иҺ·еҸ–Listпјҡ

<%
    List<String> msgs2 = Arrays.asList(msgs1);
    request.setAttribute("msgs2", msgs2);
%>

<%--el иҺ·еҸ– list--%>
<%=
((List<String>)request.getAttribute("msgs2")).get(0)
%><br>

${msgs2}<br>
${msgs2[1]}

иҺ·еҸ–Mapпјҡ

<%
    Map<String, Object> map = new HashMap<>();
    map.put("username", "root");
    map.put("password", "1234");
    session.setAttribute("map", map);
%>

<%--el иҺ·еҸ– map--%>
<%=
((HashMap<String, Object>)session.getAttribute("map")).get("username")
%><br>

${map.username}<br>
${map.password}

иҺ·еҸ–жҷ®йҖҡе®һдҪ“еҜ№иұЎпјҡ

<%
    User user = new User();
    user.setId(111);
    application.setAttribute("user", user);
%>

<%--el иҺ·еҸ– javaеҜ№иұЎ--%>
<%=
((User)application.getAttribute("user")).getId()
%><br>

${user}
${user.id}
