1. еҠЁжҖҒSQLгҖҗвҳ…гҖ‘

MyBatisзҡ„жҳ е°„ж–Үд»¶дёӯж”ҜжҢҒеңЁеҹәзЎҖSQLдёҠж·»еҠ дёҖдәӣйҖ»иҫ‘ж“ҚдҪңпјҢе№¶еҠЁжҖҒжӢјжҺҘжҲҗе®Ңж•ҙзҡ„SQLд№ӢеҗҺеҶҚжү§иЎҢпјҢд»ҘиҫҫеҲ°SQLеӨҚз”ЁгҖҒз®ҖеҢ–зј–зЁӢзҡ„ж•ҲжһңгҖӮ

1.1 < sql >

  • е®ҡд№үе…¬е…ұзҡ„ sql зүҮж®ө
<mapper namespace="com.mybatis.part2.dynamic.BookDao">
    <sql id="BOOKS_FIELD"> <!-- е®ҡд№үе…¬е…ұзҡ„ SQL зүҮж®ө -->
        SELECT id,name,author,publish,sort
    </sql>

    <select id="selectBookByCondition" resultType="com.mybatis.part2.dynamic.Book">
        <include refid="BOOKS_FIELD" /> <!-- йҖҡиҝҮIDеј•з”ЁSQLзүҮж®ө -->
        FROM t_books
    </select>
</mapper>

1.2 < where >

<select id="selectBookByCondition" resultType="com.mybatis.part2.dynamic.Book">
    SELECT id , name , author , publish , sort
    FROM t_books
    <where> 
        <if test="id != null"> <!-- WHEREпјҢдјҡиҮӘеҠЁеҝҪз•ҘеүҚеҗҺзјҖпјҲеҰӮпјҡand | orпјү -->
            id = #{id}
        </if>
        <if test="name != null">
            and name = #{name}
        </if>
        <if test="author != null">
            and author = #{author}
        </if>
        <if test="publish != null">
            and publish = #{publish}
        </if>
        <if test="sort != null">
            and sort = #{sort}
        </if>
    </where>
</select>

1.3 < set >

<update id="updateBookByCondition">
    UPDATE t_books
    <set>
        <if test="name != null">
            name = #{name} ,
        </if>
        <if test="author != null">
            author = #{author} ,
        </if>
        <if test="publish != null">
            publish = #{publish} ,
        </if>
        <if test="sort != null"> <!-- жңҖеҗҺдёҖдёӘйҖ—еҸ·йңҖиҰҒзңҒз•Ҙ -->
            sort = #{sort}
        </if>
    </set>
    WHERE id = #{id}
</update>

1.4 < trim >

<trim prefix="" suffix="" prefixOverrides="" suffixOverrides=""> д»Јжӣҝ <where> гҖҒ<set>

  • prefix еңЁtrimж ҮзӯҫеҶ…sqlиҜӯеҸҘеҠ дёҠеүҚзјҖпјҢеҰӮ whereгҖҒset
  • suffix еңЁtrimж ҮзӯҫеҶ…sqlиҜӯеҸҘеҠ дёҠеҗҺзјҖпјҢеҰӮжҸ’е…Ҙж—¶зҡ„ )
  • prefixOverridesпјҢеҝҪз•ҘеүҚзјҖпјҢйңҖжүӢеҠЁеҶҷеүҚзјҖпјҢеҰӮ andгҖҒor
  • suffixOverridesпјҢеҝҪз•ҘеҗҺзјҖпјҢйңҖжүӢеҠЁеҶҷеҗҺзјҖпјҢеҰӮ ,
<select id="selectBookByCondition" resultType="com.mybatis.day2.dynamic.Book">
    SELECT id,name,author,publish,sort FROM t_books
    <trim prefix="WHERE" prefixOverrides="AND|OR"> <!-- еўһеҠ WHEREеүҚзјҖпјҢиҮӘеҠЁеҝҪз•ҘеүҚзјҖ -->
        <if test="id != null">
            and id = #{id}
        </if>
        <if test="name != null">
            and name = #{name}
        </if>
        <if test="author != null">
            and author = #{author}
        </if>
        <if test="publish != null">
            and publish = #{publish}
        </if>
        <if test="sort != null">
            and sort = #{sort}
        </if>
    </trim>
</select>
<update id="updateBookByCondition">
    UPDATE t_books
    <trim prefix="SET" suffixOverrides=","> <!-- еўһеҠ SETеүҚзјҖпјҢиҮӘеҠЁеҝҪз•ҘеҗҺзјҖ -->
        <if test="name != null">
            name = #{name},
        </if>
        <if test="author != null">
            author = #{author},
        </if>
        <if test="publish != null">

2. зј“еӯҳпјҲCacheпјүгҖҗвҳ…гҖ‘

еҶ…еӯҳдёӯзҡ„дёҖеқ—еӯҳеӮЁз©әй—ҙпјҢжңҚеҠЎдәҺжҹҗдёӘеә”з”ЁзЁӢеәҸпјҢж—ЁеңЁе°Ҷйў‘з№ҒиҜ»еҸ–зҡ„ж•°жҚ®дёҙж—¶дҝқеӯҳеңЁеҶ…еӯҳдёӯпјҢдҫҝдәҺдәҢж¬Ўеҝ«йҖҹи®ҝй—®гҖӮ

еҰӮ redis зј“еӯҳжңҚеҠЎеҷЁпјҢжҜҸз§’иҜ»еҶҷ 10w ж•°жҚ®пјҢж•ҲзҺҮжҜ”иҫғй«ҳгҖӮ

ж— зј“еӯҳ
з”ЁжҲ·еңЁи®ҝй—®зӣёеҗҢж•°жҚ®ж—¶пјҢйңҖиҰҒеҸ‘иө·еӨҡж¬ЎеҜ№ж•°жҚ®еә“зҡ„зӣҙжҺҘи®ҝй—®пјҢдә§з”ҹеӨ§йҮҸIOгҖҒиҜ»еҶҷзЎ¬зӣҳж“ҚдҪңпјҢж•ҲзҺҮдҪҺдёӢ
жңүзј“еӯҳ
йҰ–ж¬Ўи®ҝй—®ж—¶пјҢжҹҘиҜўж•°жҚ®еә“пјҢе°Ҷж•°жҚ®еӯҳеӮЁеҲ°зј“еӯҳдёӯпјӣ
еҶҚж¬Ўи®ҝй—®ж—¶пјҢзӣҙжҺҘи®ҝй—®зј“еӯҳпјҢеҮҸе°‘IOгҖҒзЎ¬зӣҳиҜ»еҶҷж¬Ўж•°гҖҒжҸҗй«ҳж•ҲзҺҮ

2.1 дёҖзә§зј“еӯҳ

SqlSessionзә§еҲ«зҡ„зј“еӯҳпјҢеҗҢдёҖдёӘSqlSessionзҡ„еҸ‘иө·еӨҡж¬ЎеҗҢжһ„жҹҘиҜўпјҢдјҡе°Ҷж•°жҚ®дҝқеӯҳеңЁдёҖзә§зј“еӯҳдёӯгҖӮ

  • дёҖж—Ұжү§иЎҢдәҶ commit(), close(), clearCache()зӯүж–№жі•пјҢдёҖзә§зј“еӯҳе°Ҷжё…з©әгҖӮ

жіЁж„Ҹпјҡж— йңҖд»»дҪ•й…ҚзҪ®пјҢй»ҳи®ӨејҖеҗҜдёҖзә§зј“еӯҳгҖӮ
