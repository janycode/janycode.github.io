1. еј•иЁҖ

1.1 ORMжЎҶжһ¶д»Ӣз»Қ

ORMпјҲObject Relational MappingпјүеҜ№иұЎе…ізі»жҳ е°„пјҢе°ҶзЁӢеәҸдёӯзҡ„дёҖдёӘеҜ№иұЎдёҺиЎЁдёӯзҡ„дёҖиЎҢж•°жҚ®дёҖдёҖеҜ№еә”гҖӮ

ORMжЎҶжһ¶жҸҗдҫӣдәҶжҢҒд№…еҢ–зұ»дёҺиЎЁзҡ„жҳ е°„е…ізі»пјҢеңЁиҝҗиЎҢж—¶еҸӮз…§жҳ е°„ж–Үд»¶зҡ„дҝЎжҒҜпјҢжҠҠеҜ№иұЎжҢҒд№…еҢ–еҲ°ж•°жҚ®еә“дёӯгҖӮ

1.2 дҪҝз”ЁJDBCе®ҢжҲҗORMж“ҚдҪңзҡ„зјәзӮ№

  • еӯҳеңЁеӨ§йҮҸзҡ„еҶ—дҪҷд»Јз ҒгҖӮ

  • жүӢе·ҘеҲӣе»ә ConnectionгҖҒStatement зӯүгҖӮ

  • жүӢе·Ҙе°Ҷз»“жһңйӣҶе°ҒиЈ…жҲҗе®һдҪ“еҜ№иұЎгҖӮ

  • жҹҘиҜўж•ҲзҺҮдҪҺпјҢжІЎжңүеҜ№ж•°жҚ®и®ҝй—®иҝӣиЎҢиҝҮдјҳеҢ–пјҲNot CacheпјүгҖӮ

2. MyBatisжЎҶжһ¶

package com.mybatis.part1.basic;

import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import org.junit.Test;

import java.io.IOException;
import java.io.InputStream;

public class HelloMyBatis {

    @Test
    public void test1() throws IOException {
        //1.иҺ·еҫ—иҜ»еҸ–MyBatisй…ҚзҪ®ж–Үд»¶зҡ„жөҒеҜ№иұЎпјҲжіЁж„Ҹ Resources еҜјеҢ…дёә org.apache.ibatis.ioпјү
        InputStream is = Resources.getResourceAsStream("mybatis-config.xml");

        //2.жһ„е»әSqlSessionиҝһжҺҘеҜ№иұЎзҡ„е·ҘеҺӮ
        SqlSessionFactory factory = new SqlSessionFactoryBuilder().build(is);

        //3.йҖҡиҝҮе·ҘеҺӮиҺ·еҫ— sql дјҡиҜқеҜ№иұЎпјҲиҜҘеҜ№иұЎеҸҜд»ҘзӣҙжҺҘж“ҚдҪңж•°жҚ®еә“пјү
        SqlSession sqlSession = factory.openSession();

        //4.йҖҡиҝҮиҝһжҺҘеҜ№иұЎиҺ·еҫ—жҺҘеҸЈе®һзҺ°зұ»еҜ№иұЎпјҢе№¶и°ғз”Ёж–№жі•жү§иЎҢ
        User user = sqlSession.getMapper(UserDao.class).selectUserById(1);
        System.out.println(user);
    }
}
