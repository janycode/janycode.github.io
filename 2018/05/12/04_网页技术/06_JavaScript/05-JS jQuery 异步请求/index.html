05-JS jQuery ејӮжӯҘиҜ·жұӮ

>> jQuery еҸӮиҖғжүӢеҶҢ

jQueryпјҢжҳҜдёҖдёӘ JavaScript и„ҡжң¬зұ»еә“ж–Үд»¶пјҢз®ҖеҢ–дәҶJSзј–зЁӢгҖӮ
JQuery жҳҜ2006е№ҙ8жңҲиҜһз”ҹзҡ„дёҖдёӘејҖжәҗйЎ№зӣ®пјҢеҮӯеҖҹз®ҖжҙҒзҡ„иҜӯжі•е’Ңи·Ёе№іеҸ°зҡ„е…је®№жҖ§пјҢжһҒеӨ§ең°з®ҖеҢ–дәҶ JavaScript ејҖеҸ‘йҒҚеҺҶ HTML ж–ҮжЎЈгҖҒж“ҚдҪң DOM гҖҒеӨ„зҗҶдәӢд»¶гҖҒжү§иЎҢеҠЁз”»е’ҢејҖеҸ‘ Ajax зҡ„ж“ҚдҪңпјҢе…¶зӢ¬зү№иҖҢеҸҲдјҳйӣ…зҡ„д»Јз ҒйЈҺж јж”№еҸҳдәҶ JavaScript зЁӢеәҸе‘ҳзҡ„и®ҫи®ЎжҖқи·Ҝе’Ңзј–еҶҷзЁӢеәҸзҡ„ж–№ејҸгҖӮ

1. jQuery еҜје…Ҙж–№ејҸ

1.1 дёӢиҪҪеҜје…Ҙ

д»Һе®ҳзҪ‘дёӢиҪҪ jQuery еә“пјҡhttps://jquery.com/download/

еҜје…Ҙ(/web/js/ зӣ®еҪ•дёӢ)пјҡ

<script src="js/jquery-1.10.2.min.js"></script>

1.2 CDNиҪҪе…Ҙ

CDNпјҢеҶ…е®№еҲҶеҸ‘зҪ‘з»ңгҖӮ
е®ҳзҪ‘ CDN йҖүжӢ© minified (еҺӢзј©зүҲ) еҚіеҸҜпјҡhttps://code.jquery.com/
иҪҪе…Ҙпјҡ

е®ҳзҪ‘пјҡ
<script src="https://code.jquery.com/jquery-3.5.1.min.js"></script>

е…¶д»–з«ҷзӮ№ CDN пјҡ

зҷҫеәҰпјҡ
<script src="https://apps.bdimg.com/libs/jquery/2.1.4/jquery.min.js"></script>
и°·жӯҢпјҡ
<script src="https://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
еҫ®иҪҜпјҡ

еҸӮиҖғжүӢеҶҢпјҡhttps://www.w3school.com.cn/jquery/jquery_ref_ajax.asp
йҖҡиҝҮ jQuery AJAX ж–№жі•пјҢжӮЁиғҪеӨҹдҪҝз”Ё HTTP Get е’Ң HTTP Post д»ҺиҝңзЁӢжңҚеҠЎеҷЁдёҠиҜ·жұӮж–Үжң¬гҖҒHTMLгҖҒXML жҲ– JSONпјҢеҗҢж—¶жӮЁиғҪеӨҹжҠҠиҝҷдәӣеӨ–йғЁж•°жҚ®зӣҙжҺҘиҪҪе…ҘзҪ‘йЎөзҡ„иў«йҖүе…ғзҙ дёӯгҖӮ

  • $(selector).load(url,data,callback) жҠҠиҝңзЁӢж•°жҚ®еҠ иҪҪеҲ°иў«йҖүзҡ„е…ғзҙ дёӯ
  • $.ajax(options) жҠҠиҝңзЁӢж•°жҚ®еҠ иҪҪеҲ° XMLHttpRequest еҜ№иұЎдёӯ
  • $.get(url,data,callback,type) дҪҝз”Ё HTTP GET жқҘеҠ иҪҪиҝңзЁӢж•°жҚ®
  • $.post(url,data,callback,type) дҪҝз”Ё HTTP POST жқҘеҠ иҪҪиҝңзЁӢж•°жҚ®
  • $.getJSON(url,data,callback) дҪҝз”Ё HTTP GET жқҘеҠ иҪҪиҝңзЁӢ JSON ж•°жҚ®
  • $.getScript(url,callback) еҠ иҪҪе№¶жү§иЎҢиҝңзЁӢзҡ„ JavaScript ж–Үд»¶

еҸӮж•°иҜҙжҳҺпјҡ
@selector, jQuery е…ғзҙ йҖүжӢ©еҷЁиҜӯжі•
@url, иў«еҠ иҪҪзҡ„ж•°жҚ®зҡ„ URLпјҲең°еқҖпјү
@data, еҸ‘йҖҒеҲ°жңҚеҠЎеҷЁзҡ„ж•°жҚ®зҡ„й”®/еҖјеҜ№иұЎ

3.3 ajax

$.ajax([settings]) жҳҜдҪҺеұӮзә§ AJAX еҮҪж•°зҡ„иҜӯжі•гҖӮ
$.ajax жҸҗдҫӣдәҶжҜ”й«ҳеұӮзә§еҮҪж•°жӣҙеӨҡзҡ„еҠҹиғҪпјҢдҪҶжҳҜеҗҢж—¶д№ҹжӣҙйҡҫдҪҝз”ЁгҖӮ

  • asyncпјҢи®ҫзҪ®жҳҜеҗҰдёәејӮжӯҘиҜ·жұӮ
  • contentTypeпјҢи®ҫзҪ®иҜ·жұӮеӨҙContent-TypeпјҢй»ҳи®ӨеҖјпјҡвҖқapplication/x-www-form-urlencodedвҖқ

  • errorпјҢиҜ·жұӮеӨұиҙҘзҡ„еӣһи°ғеҮҪж•°
  • successпјҢиҜ·жұӮжҲҗеҠҹзҡ„еӣһи°ғеҮҪж•°

@WebServlet(name = "CheckUsernameServlet", urlPatterns = "/checkUsername")
public class CheckUsernameServlet extends HttpServlet {
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        request.setCharacterEncoding("utf-8");
        String username = request.getParameter("username");
        UserService userService = new UserServiceImpl();
        try {
            boolean flag = userService.checkUsername(username);
            String msg = flag ? "з”ЁжҲ·еҗҚеҸҜз”Ё" : "з”ЁжҲ·еҗҚе·ІеӯҳеңЁ";
            Map<String, Object> map = new HashMap<>();
            map.put("flag", flag);
            map.put("msg", msg);

            JsonUtils.writeJsonStr(response, map);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doPost(request, response);
    }
}
