  • eq() ж–№жі•иҝ”еӣһиў«йҖүе…ғзҙ дёӯеёҰжңүжҢҮе®ҡзҙўеј•еҸ·зҡ„е…ғзҙ гҖӮ

  • зҙўеј•еҸ·д»Һ 0 ејҖе§ӢпјҢеӣ жӯӨйҰ–дёӘе…ғзҙ зҡ„зҙўеј•еҸ·жҳҜ 0 иҖҢдёҚжҳҜ 1гҖӮйҖүеҸ–з¬¬дәҢдёӘp е…ғзҙ пјҲзҙўеј•еҸ· 1пјүпјҡ

$(document).ready(function(){
  $("p").eq(1);
});

  • filter() ж–№жі•е…Ғи®ёи§„е®ҡдёҖдёӘж ҮеҮҶгҖӮдёҚеҢ№й…ҚиҝҷдёӘж ҮеҮҶзҡ„е…ғзҙ дјҡиў«д»ҺйӣҶеҗҲдёӯеҲ йҷӨпјҢеҢ№й…Қзҡ„е…ғзҙ дјҡиў«иҝ”еӣһгҖӮ

  • иҝ”еӣһеёҰжңүзұ»еҗҚ вҖңurlвҖқ зҡ„жүҖжңүp е…ғзҙ пјҡ

$(document).ready(function(){
  $("p").filter(".url");
});

  • not() ж–№жі•иҝ”еӣһдёҚеҢ№й…Қж ҮеҮҶзҡ„жүҖжңүе…ғзҙ гҖӮ

  • жҸҗзӨәпјҡnot() ж–№жі•дёҺ filter() зӣёеҸҚгҖӮ

  • иҝ”еӣһдёҚеёҰжңүзұ»еҗҚ вҖңurlвҖқ зҡ„жүҖжңүpе…ғзҙ пјҡ

$(document).ready(function(){
  $("p").not(".url");
});

7.1 jQuery AJAXз®Җд»Ӣ

  • AJAX = ејӮжӯҘ JavaScript е’Ң XMLпјҲAsynchronous JavaScript and XMLпјүгҖӮ

  • з®Җзҹӯең°иҜҙпјҢеңЁдёҚйҮҚиҪҪж•ҙдёӘзҪ‘йЎөзҡ„жғ…еҶөдёӢпјҢAJAX йҖҡиҝҮеҗҺеҸ°еҠ иҪҪж•°жҚ®пјҢе№¶еңЁзҪ‘йЎөдёҠиҝӣиЎҢжҳҫзӨәгҖӮ

  • дҪҝз”Ё AJAX зҡ„еә”з”ЁзЁӢеәҸжЎҲдҫӢпјҡи°·жӯҢең°еӣҫгҖҒи…ҫи®Ҝеҫ®еҚҡгҖҒдјҳй…·и§Ҷйў‘гҖҒдәәдәәзҪ‘зӯүзӯүгҖӮ

7.2 getе’Ңpostж–№жі•

  • $.get() ж–№жі•йҖҡиҝҮ HTTP GET иҜ·жұӮд»ҺжңҚеҠЎеҷЁдёҠиҜ·жұӮж•°жҚ®гҖӮ
    • $.get(url, callback);
      • еҝ…йңҖзҡ„ URL еҸӮж•°и§„е®ҡеёҢжңӣиҜ·жұӮзҡ„ URLгҖӮ
    • еҸҜйҖүзҡ„ callback еҸӮж•°жҳҜиҜ·жұӮжҲҗеҠҹеҗҺжүҖжү§иЎҢзҡ„еҮҪж•°еҗҚгҖӮ
      • дҪҝз”Ё $.get() ж–№жі•д»ҺжңҚеҠЎеҷЁдёҠзҡ„дёҖдёӘж–Үд»¶дёӯеҸ–еӣһж•°жҚ®пјҡ
$("button").click(function(){
  $.get("demo_test.jsp?paramer=xxx",function(data){
    alert("ж•°жҚ®: " + data );
  });
});
  • $.post() ж–№жі•йҖҡиҝҮ HTTP POST иҜ·жұӮд»ҺжңҚеҠЎеҷЁдёҠиҜ·жұӮж•°жҚ®гҖӮ
    • $.post(url, data, callback);
    • еҝ…йңҖзҡ„ URL еҸӮж•°и§„е®ҡеёҢжңӣиҜ·жұӮзҡ„ URLгҖӮ
    • еҸҜйҖүзҡ„ data еҸӮж•°и§„е®ҡиҝһеҗҢиҜ·жұӮеҸ‘йҖҒзҡ„ж•°жҚ®гҖӮ
    • еҸҜйҖүзҡ„ callback еҸӮж•°жҳҜиҜ·жұӮжҲҗеҠҹеҗҺжүҖжү§иЎҢзҡ„еҮҪж•°еҗҚгҖӮ
  • дҪҝз”Ё $.post() иҝһеҗҢиҜ·жұӮдёҖиө·еҸ‘йҖҒж•°жҚ®пјҡ
$("button").click(function(){
    $.post("/try/ajax/demo_test_post.jsp",
    {
        name:"зҷҫеәҰ",
        url:"http://www.baidu.com"
    },
        function(data){
        alert("ж•°жҚ®: \n" + data );
    });
});

7.3 ajax()ж–№жі•

  • $.ajax([settings])
  • jQuery еә•еұӮ AJAX е®һзҺ°гҖӮз®ҖеҚ•жҳ“з”Ёзҡ„й«ҳеұӮе®һзҺ°и§Ғ $.get, $.post зӯүгҖӮ$.ajax() иҝ”еӣһе…¶еҲӣе»әзҡ„ XMLHttpRequest еҜ№иұЎгҖӮеӨ§еӨҡж•°жғ…еҶөдёӢж— йңҖзӣҙжҺҘж“ҚдҪңиҜҘеҮҪж•°пјҢйҷӨйқһйңҖиҰҒж“ҚдҪңдёҚеёёз”Ёзҡ„йҖүйЎ№пјҢд»ҘиҺ·еҫ—жӣҙеӨҡзҡ„зҒөжҙ»жҖ§гҖӮ
  • жңҖз®ҖеҚ•зҡ„жғ…еҶөдёӢпјҢ$.ajax()еҸҜд»ҘдёҚеёҰд»»дҪ•еҸӮж•°зӣҙжҺҘдҪҝз”ЁгҖӮ

8.1 jQuery noConflictж–№жі•

  • жӯЈеҰӮе·Із»ҸдәҶи§ЈеҲ°зҡ„пјҢjQuery дҪҝз”Ё $ з¬ҰеҸ·дҪңдёә jQuery зҡ„з®ҖеҶҷгҖӮ

  • еҰӮжһңе…¶д»– JavaScript жЎҶжһ¶д№ҹдҪҝз”Ё $ з¬ҰеҸ·дҪңдёәз®ҖеҶҷжҖҺд№ҲеҠһпјҹ

  • е…¶д»–дёҖдәӣ JavaScript жЎҶжһ¶еҢ…жӢ¬пјҡMooToolsгҖҒBackboneгҖҒSammyгҖҒCappuccinoгҖҒKnockoutгҖҒJavaScript MVCгҖҒGoogle Web ToolkitгҖҒGoogle ClosureгҖҒEmberгҖҒBatman д»ҘеҸҠ Ext JSгҖӮ

  • е…¶дёӯжҹҗдәӣжЎҶжһ¶д№ҹдҪҝз”Ё $ з¬ҰеҸ·дҪңдёәз®ҖеҶҷпјҲе°ұеғҸ jQueryпјүпјҢеҰӮжһңеңЁз”Ёзҡ„дёӨз§ҚдёҚеҗҢзҡ„жЎҶжһ¶жӯЈеңЁдҪҝз”ЁзӣёеҗҢзҡ„з®ҖеҶҷз¬ҰеҸ·пјҢжңүеҸҜиғҪеҜјиҮҙи„ҡжң¬еҒңжӯўиҝҗиЎҢгҖӮ

  • jQuery зҡ„еӣўйҳҹиҖғиҷ‘еҲ°дәҶиҝҷдёӘй—®йўҳпјҢе№¶е®һзҺ°дәҶ noConflict() ж–№жі•гҖӮ

  • noConflict() ж–№жі•дјҡйҮҠж”ҫеҜ№ $ ж ҮиҜҶз¬Ұзҡ„жҺ§еҲ¶пјҢиҝҷж ·е…¶д»–и„ҡжң¬е°ұеҸҜд»ҘдҪҝз”Ёе®ғдәҶгҖӮ

  • еҪ“з„¶пјҢд»Қз„¶еҸҜд»ҘйҖҡиҝҮе…ЁеҗҚжӣҝд»Јз®ҖеҶҷзҡ„ж–№ејҸжқҘдҪҝз”Ё jQueryпјҡ

$.noConflict();
jQuery(document).ready(function(){
  jQuery("button").click(function(){
    jQuery("p").text("jQuery д»Қз„¶еңЁе·ҘдҪң!");
  });
});

д№ҹеҸҜд»ҘеҲӣе»әз®ҖеҶҷгҖӮnoConflict() еҸҜиҝ”еӣһеҜ№ jQuery зҡ„еј•з”ЁпјҢеҸҜд»ҘжҠҠе®ғеӯҳе…ҘеҸҳйҮҸпјҢд»ҘдҫӣзЁҚеҗҺдҪҝз”ЁгҖӮиҜ·зңӢиҝҷдёӘдҫӢеӯҗпјҡ

var jq = $.noConflict();
jq(document).ready(function(){
