<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title></title>
    </head>
    <body>
        地址：
        <select id="province" onchange="showCity()">
        </select> 省&nbsp;
        <select id="city" onchange="showCounty()">
        </select> 市&nbsp;
        <select id="county">
        </select> 区/县
    </body>
    <script type="text/javascript">
        var city = /* 三级联城市 JSON 字符串(本文下方) */;

        var citys = JSON.parse(city);
        console.log(citys); // 转 js 对象
        console.log(citys.citylist[4].p); // eg:获取省份
        console.log(citys.citylist[4].c[0].n) // eg:获取城市
        console.log(citys.citylist[4].c[0].a[0].s) // eg:获取区/县
        //var provName = citys.citylist[province.selectedIndex].p;

        //var counName = citys.citylist[province.selectedIndex].c[city.selectedIndex].a[county.selectedIndex].s;

        var province = document.getElementById("province");
        var city = document.getElementById("city");
        var county = document.getElementById("county");

        // 清空"省份"
        // var provChild = province.childNodes;
        // for (var i = 0; i < provChild.length;) {
        //     province.removeChild(provChild[i]);
        // }
        // 页面加载完默认先把省份添加
        var provArr = citys.citylist;
        for (var i = 0; i < provArr.length; i++) {
            var op = document.createElement("option");
            op.innerHTML = provArr[i].p;
            province.appendChild(op);
        }
        showCity();
        //showCounty();

        function showCity() {
            // 清空上一个选中的级联城市
            var cityChild = city.childNodes;
            for (var i = 0; i < cityChild.length;) {
                city.removeChild(cityChild[i]);
            }

            // 添加城市 到 对应省份
            var cityArr = citys.citylist[province.selectedIndex].c;
            for (var i = 0; i < cityArr.length; i++) {
                var op = document.createElement("option");
                op.innerHTML = cityArr[i].n;
                city.appendChild(op);
            }

            // 级联的 区县 跟随变化
            showCounty();
        }

        function showCounty() {
            // 清空上一个选中的级联区县
            var countyChild = county.childNodes;
            for (var i = 0; i < countyChild.length;) {
                county.removeChild(countyChild[i]);
            }

            // 添加区县 到 对应城市
            var countyArr = citys.citylist[province.selectedIndex].c[city.selectedIndex].a;
            for (var i = 0; i < countyArr.length; i++) {
                var op = document.createElement("option");
                op.innerHTML = countyArr[i].s;
                county.appendChild(op);
            }
        }
    </script>
