02-еӯ—иҠӮжөҒиҜ»еҶҷ

02_зј–зЁӢиҜӯиЁҖ > 01_Java > 01_JavaSE > 06_ж–Үд»¶е’ҢжөҒ
дҪңиҖ…:Jerry ж—¶й—ҙ:2016-04-28 22:03:43 и®ҝй—®:

1. еӯ—иҠӮиҫ“е…Ҙиҫ“еҮәзұ»

2. жҠҪиұЎзҲ¶зұ»пјҡInputStream & OutputStream

InputStreamеӯ—иҠӮиҫ“е…ҘжөҒпјҡ

public abstract class InputStream
    extends Object
    implements Closeable

иҝҷдёӘжҠҪиұЎзұ»жҳҜиЎЁзӨәиҫ“е…Ҙеӯ—иҠӮжөҒзҡ„жүҖжңүзұ»зҡ„и¶…зұ»гҖӮйңҖиҰҒе®ҡд№үInputStreamеӯҗзұ»зҡ„еә”з”ЁInputStreamеҝ…йЎ»е§Ӣз»ҲжҸҗдҫӣдёҖз§Қиҝ”еӣһиҫ“е…Ҙзҡ„дёӢдёҖдёӘеӯ—иҠӮзҡ„ж–№жі•гҖӮ

abstract int read()
д»Һиҫ“е…ҘжөҒиҜ»еҸ–ж•°жҚ®зҡ„дёӢдёҖдёӘеӯ—иҠӮгҖӮ
int read(byte[] b)
д»Һиҫ“е…ҘжөҒиҜ»еҸ–дёҖдәӣеӯ—иҠӮж•°пјҢе№¶е°Ҷе®ғд»¬еӯҳеӮЁеҲ°зј“еҶІеҢә b гҖӮ
int read(byte[] b, int off, int len)
д»Һиҫ“е…ҘжөҒиҜ»еҸ–жңҖеӨҡ lenеӯ—иҠӮзҡ„ж•°жҚ®еҲ°дёҖдёӘеӯ—иҠӮж•°з»„гҖӮ

OutputStreamеӯ—иҠӮиҫ“еҮәжөҒпјҡ

public abstract class OutputStream
    extends Object
    implements Closeable, Flushable

иҝҷдёӘжҠҪиұЎзұ»жҳҜиЎЁзӨәеӯ—иҠӮиҫ“еҮәжөҒзҡ„жүҖжңүзұ»зҡ„и¶…зұ»гҖӮ иҫ“еҮәжөҒжҺҘж”¶иҫ“еҮәеӯ—иҠӮе№¶е°Ҷе…¶еҸ‘йҖҒеҲ°жҹҗдёӘжҺҘж”¶еҷЁгҖӮйңҖиҰҒе®ҡд№үOutputStreamеӯҗзұ»зҡ„еә”з”ЁOutputStreamеҝ…йЎ»иҮіе°‘жҸҗдҫӣдёҖдёӘеҶҷе…ҘдёҖдёӘеӯ—иҠӮиҫ“еҮәзҡ„ж–№жі•гҖӮ

void write(byte[] b)
е°Ҷ b.lengthеӯ—иҠӮд»ҺжҢҮе®ҡзҡ„еӯ—иҠӮж•°з»„еҶҷе…ҘжӯӨиҫ“еҮәжөҒгҖӮ
void write(byte[] b, int off, int len)
д»ҺжҢҮе®ҡзҡ„еӯ—иҠӮж•°з»„еҶҷе…Ҙ lenдёӘеӯ—иҠӮпјҢд»ҺеҒҸз§» offејҖе§Ӣиҫ“еҮәеҲ°жӯӨиҫ“еҮәжөҒгҖӮ
abstract void write(int b)
е°ҶжҢҮе®ҡзҡ„еӯ—иҠӮеҶҷе…ҘжӯӨиҫ“еҮәжөҒгҖӮ

2.1 е®һзҺ°зұ»пјҡеӯ—иҠӮиҠӮзӮ№жөҒ FileOutputStream & FileInputStream

  • FileOutputStreamзұ»пјҲеӯ—иҠӮиҠӮзӮ№иҫ“еҮәжөҒпјүпјҡ
public class FileOutputStream
    extends OutputStream

ж ёеҝғж–№жі•пјҡ

void write(int b)
е°ҶжҢҮе®ҡзҡ„еӯ—иҠӮеҶҷе…ҘжӯӨж–Үд»¶иҫ“еҮәжөҒгҖӮ
void write(byte[] b)
е°Ҷ b.lengthдёӘеӯ—иҠӮд»ҺжҢҮе®ҡзҡ„еӯ—иҠӮж•°з»„еҶҷе…ҘжӯӨж–Үд»¶иҫ“еҮәжөҒгҖӮ
void write(byte[] b, int off, int len)
е°Ҷ lenеӯ—иҠӮд»ҺдҪҚдәҺеҒҸз§»йҮҸ offзҡ„жҢҮе®ҡеӯ—иҠӮж•°з»„еҶҷе…ҘжӯӨж–Үд»¶иҫ“еҮәжөҒгҖӮ

жһ„йҖ ж–№жі•пјҡ

FileOutputStream(File file)
еҲӣе»әж–Үд»¶иҫ“еҮәжөҒд»ҘеҶҷе…Ҙз”ұжҢҮе®ҡзҡ„ FileеҜ№иұЎиЎЁзӨәзҡ„ж–Үд»¶гҖӮ
FileOutputStream(File file, boolean append)
еҲӣе»әж–Үд»¶иҫ“еҮәжөҒд»ҘеҶҷе…Ҙз”ұжҢҮе®ҡзҡ„ FileеҜ№иұЎиЎЁзӨәзҡ„ж–Үд»¶пјҢappend==trueиҝҪеҠ пјҢдёҚиҰҶзӣ–гҖӮ
FileOutputStream(FileDescriptor fdObj)
еҲӣе»әж–Үд»¶иҫ“еҮәжөҒд»ҘеҶҷе…ҘжҢҮе®ҡзҡ„ж–Үд»¶жҸҸиҝ°з¬ҰпјҢиЎЁзӨәдёҺж–Үд»¶зі»з»ҹдёӯе®һйҷ…ж–Үд»¶зҡ„зҺ°жңүиҝһжҺҘгҖӮ
FileOutputStream(String name)
еҲӣе»әж–Үд»¶иҫ“еҮәжөҒд»ҘжҢҮе®ҡзҡ„еҗҚз§°еҶҷе…Ҙж–Үд»¶гҖӮ

class Student implements Serializable {
      private static final long serialVersionUID = 1L;
      /*static*/String name; // staticдёәзұ»жүҖжңүпјҢдјҡеҪұе“ҚиҜҘеұһжҖ§еҖјеҸҳдёәдёҖдёӘзӣёеҗҢзҡ„еҖј
      Integer age;
      String sex;
      transient Double score; // transientдҝ®йҘ°зҡ„еұһжҖ§дёҚеҸӮдёҺеәҸеҲ—еҢ–пјҢеқҮдёәnull
      public Student() {
            super();
      }
      public Student(String name, Integer age, String sex, Double  score) {
            super();
            this.name = name;
            this.age = age;
            this.sex = sex;
            this.score = score;
      }
      @Override
      public String toString() {
            return "Student [name=" + name + ", age=" + age + ", sex="  + sex + ", score=" + score + "]";
      }
}
